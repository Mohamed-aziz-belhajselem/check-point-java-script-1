const GUEST_LIST = {
	Randy: "Germany",
	Karla: "France",
	Wendy: "Japan",
	Norman: "England",
	Sam: "Argentina"
}

function greeting(name) {
	var b = Object.keys(GUEST_LIST);
	if (b.indexOf(name) == -1){
		return "Hi! I'm a guest.";
	} else {
		return b.map(function(key,index){
		if (key == name || key == index) {
			return `Hi! I\'m ${key}, and I\'m from ${GUEST_LIST[key]}.`
		}
	}).join('');
	}
}
